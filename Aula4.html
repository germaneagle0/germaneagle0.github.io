from numpy import sign

def pesquisar intervalos(func, a, b, dx):
    x1 = a; f1 = func(a)
    x2 = a + dx; f2 = func(b)

    raizes = []
    while x2 <= b:
        if (f1 == 0):
            raizes.append(x1,x1)
        if sign(f1) != sign(f2):
            raizes.append(x1, x2)
        x1 = x2 ; f1 = f2
        x2 += dx ; f2 = func(x2)
    return raizes
